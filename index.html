const apiUrl = 'https://marketplace-api.ortak.me/api/v1/collections/6/nfts'; const corsProxy = 'https://cors-anywhere.herokuapp.com/'; async function fetchData() { try { const response = await fetch(corsProxy + apiUrl); const data = await response.json(); displayData(data); } catch (error) { document.getElementById('data-container').innerText = 'Failed to load data.'; console.error('Error fetching data:', error); } } function displayData(data) { const dataContainer = document.getElementById('data-container'); dataContainer.innerHTML = `
${JSON.stringify(data, null, 2)}
`; } // Fetch and display data on page load fetchData();
